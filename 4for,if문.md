package jeongmin4;

public class main {
	final static int N=15;

	public static void main(String[] args) {
		for(int i=-N;i<=N;i++)
		{
			for (int j=-N;j<=30;j++)
			{
				if(i*i+j*j<=N*N)
				{
					System.out.print("*");
				}
				else {
					System.out.println(" ");
				}
				
			}
			System.out.println("");
			}
		}
}
